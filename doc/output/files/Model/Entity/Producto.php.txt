<?php
namespace App\Model\Entity;

use Cake\ORM\Entity;

/**
 * Producto Entity
 *
 * @property string $idProducto
 * @property string $nombreProducto
 * @property int $tipo
 * @property string $imagen
 * @property int $precio
 * @property string $fabricante
 */
class Producto extends Entity
{

    /**
     * Fields that can be mass assigned using newEntity() or patchEntity().
     *
     * Note that when '*' is set to true, this allows all unspecified fields to
     * be mass assigned. For security purposes, it is advised to set '*' to false
     * (or remove it), and explicitly make individual fields accessible as needed.
     *
     * @var array
     */
    protected $_accessible = [
        '*' => true,
        'idProducto' => false
    ];
    
     /* No Estoy  seguro que nos sea util, modifique el de la contraseÃ±a
     del lab (esta en user en esta carpeta) la idea es que le permita al 
     administrador insertar un id */
    protected function _setId($value)
    {
        if (!empty($value))
        {
            $hasher = new DefaultPasswordHasher();
            return $hasher->hash($value);
        }
    }
}
